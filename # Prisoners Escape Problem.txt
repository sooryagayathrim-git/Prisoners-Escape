# Prisoners Escape Problem

# Input the number of prisoners
n = int(input("Enter the number of prisoners: "))

# Input lock strengths
strength = []
for i in range(n):
    b = int(input(f"Enter strength of prisoner {i+1}: "))
    strength.append(b)

print("Lock strengths:", strength)

# Count prisoners who can escape
count = 0
for j in range(n):
    if j == 0:
        if strength[j] < strength[j+1]:
            count += 1
    elif j == n - 1:
        if strength[j] < strength[j-1]:
            count += 1
    else:
        if strength[j] < strength[j-1] and strength[j] < strength[j+1]:
            count += 1

print("Number of prisoners who can escape:", count)
